#include<reg52.h>			/*Подключение библиотек для работы с MCS 51*/			
#include<delay51.h>

sbit BUTTON_ON =P3^2;			/*Подключение кнопки к порту и выводу(ножки) микроконтроллера*/
sbit BUTTON_ST =P3^3;
sbit BUTTON_GW = P1^7;
 
sbit LED_RED=P1^0; 			/*Подключение кнопки к порту и выводу(ножки) светодиода
sbit LED_YELLOW=P1^1; 
sbit LED_GREEN=P1^2; 

#define YELLOWBLINK 1			/*Определение идентификатора и последовательности символов,  
#define YELLOWUP 2			/*которой будет замещаться данный идентификатор*/
#define RED	3			/*при его обнаружении в тексте программы. */
#define YELLOWDOWN 4			
#define GREEN  5
#define GREENWAVE 6


unsigned char state = YELLOWBLINK;	/*Переменная считывания текущего состояния*/
unsigned char nextState = YELLOWBLINK;  /*Переменная считывания последующего состояния*/

bit unPushed = 0;			/*Флаг удержания "Зеленой Волны" при многократном нажатии кнопки*/
					/*или программного отключения после окончания действия*/

void StateTransition();			/*Объявление функций перехода и работы с таймером*/
void delay_us(unsigned int uSeconds);	
void delay_ms(unsigned int mSeconds);	

void delay_us(unsigned int uSeconds)	/*Функция инициализации задержки в микросекундах с использованием Timer1*/
{
	uSeconds = 0xFFFF - uSeconds; 	/*Инициализация таймера-счетчика*/
					/*При 12 МГЦ "тик" таймера должен быть каждую микросекунду (CLK_FREQ / 12)*/
	TMOD |= 0x10;   		/*Включение режима работы таймера*/  
	TL1 = uSeconds;     		/*Загрузка младших битов таймера*/
	TH1 = uSeconds>>8;		/*Загрузка старших битов таймера*/		
	TR1 = 1;			/*Пуск таймера*/
	while (TF1==0);			/*Ожидание переполнения таймера-счетчика*/
	TR1 = 0;			/*Выключение таймера*/
	TF1 = 0;			/*Сброс счетчика*/
}


	
void delay_ms(unsigned int mSeconds)	/*Функция инициализации задержки в микросекундах с использованием Timer1*/
{
	while (mSeconds--)		/*с помощью значений в милисекундах*/
	{
		delay_us(1000);
	}
	
	state = nextState;		/*При завершении работы таймера переход в новое состояние*/
}


void main (void)			/*Точка входа в программу*/
{		
	while(1)			
	{	
		StateTransition();	/*Непрерывный вызов функции переходов и выходов*/
	}
}


void StateTransition()  		/*Функция переходов и выходов*/
{ 
	 while (1)
		 {
			 switch(state)	/*Описание состояний*/
			 {
				 case YELLOWBLINK:	/*Блок описания состояния*/
					 
							if(BUTTON_ON == 0) /*Нажатие кнопки соответсвтует переходу в следующее состояние*/
							{
									nextState = YELLOWUP;									
							}
						
							else 
							{
									nextState = YELLOWBLINK;
							}
						
							break;
											
					case YELLOWUP:
							
							if(BUTTON_ST == 1 || BUTTON_ON == 1)
							{
									nextState = YELLOWBLINK;
							}
							
							else if (BUTTON_ST == 0)
							{
									nextState = RED;
							}	

							if (BUTTON_GW == 1) /*Если режим "Зеленая Волна" был отключен отжатием кнопки*/
							{
									unPushed = 0;	/*Сброс флага для последующего вызова "Зеленой Волны"*/
							}
							
							break;
							
					case RED:

							if(BUTTON_ST == 1 || BUTTON_ON == 1)
							{
									nextState = YELLOWBLINK;
							}
							
							else if (BUTTON_ST == 0)
							{
									nextState = YELLOWDOWN;
							}	
							
							break;
					
					case YELLOWDOWN:

							if(BUTTON_ST == 1 || BUTTON_ON == 1)
							{
									nextState = YELLOWBLINK;
							}
							
							else if (BUTTON_ST == 0)
							{
									nextState = GREEN;
							}	
							
							if (BUTTON_GW == 0 && unPushed == 0) /*Если кнопка вызова "Зеленая волна нажата и флаг сброшен"*/
							{
									nextState = GREENWAVE; /*Переход в режим "Зеленая Волна"*/
							}
							
							break;
					
					case GREEN:

							if(BUTTON_ST == 1 || BUTTON_ON == 1)
							{
									nextState = YELLOWBLINK;
							}
							
							else if (BUTTON_ST == 0)
							{
									nextState = YELLOWUP;
							}	
							
							break;
		
						
					case GREENWAVE:	/*При неотключении кнопки вызова режима "Зеленая волна"*/
								
								unPushed = 1; /*Установка флага для перехода в следующее состояние при окончании работы режима*/
								nextState = YELLOWUP; 
							
							break;

						
					default: /*Блок описания перехода устройства при сбое*/
							nextState = YELLOWBLINK;
							break;
			}
			
			switch(nextState) /*Описание выходных значений*/
			{	
				case RED: /*Блок описания выходного значения*/
					LED_RED = 0;
					LED_YELLOW = 1;
					LED_GREEN = 1;
					delay_ms(5000); /*Вызов функции(прерывания) таймера*/
					break;
			
				case YELLOWUP:
					LED_RED = 1;
					LED_YELLOW = 0;
					LED_GREEN = 1;
					delay_ms(1000);
					break;	
			
				case YELLOWDOWN:
					LED_RED = 0;
					LED_YELLOW = 0;
					LED_GREEN = 1;
					delay_ms(1000);
					break; 
			
				case GREEN:
					LED_RED = 1;
					LED_YELLOW = 1;
					LED_GREEN = 0;
					delay_ms(5000);
					break;  
			
				case YELLOWBLINK:
					LED_RED = 1;
					LED_YELLOW = 1;
					LED_GREEN = 1;
					delay_ms(1000);
					break;  
			
				case GREENWAVE:
					LED_RED = 1;
					LED_YELLOW = 1;
					LED_GREEN = 0;
					delay_ms(15000);	
					break;  
			
				default: /*Блок описания выходных значений устройства при сбое*/
					LED_RED = 1;
					LED_YELLOW = 1;
					LED_GREEN = 1;
					break;
			}
	 }
}
